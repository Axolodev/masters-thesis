% TOC spacing configuration
\setlength{\cftbeforetoctitleskip}{0pt}  % space before TOC title
% (can be negative)
\setlength{\cftaftertoctitleskip}{1em}    % space after TOC title

% List of Tables spacing
\setlength{\cftbeforelottitleskip}{0pt}  % space before LOT title
\setlength{\cftafterlottitleskip}{1em}   % space after LOT title

% List of Figures spacing
\setlength{\cftbeforeloftitleskip}{0pt}  % space before LOF title
\setlength{\cftafterloftitleskip}{1em}   % space after LOF title

\setlength{\cftbeforechapskip}{1em}   % vertical space before chapter entries
\setlength{\cftbeforesecskip}{0.5em}    % vertical space before section entries
\setlength{\cftbeforesubsecskip}{0.25em} % vertical space before
% subsection entries

% Optional: change font size of TOC entries
\renewcommand{\cftchapfont}{\normalsize}     % chapter font size
\renewcommand{\cftchappagefont}{\normalsize} % chapter page numbers
\renewcommand{\cftsecfont}{\normalsize}           % section font size
\renewcommand{\cftsubsecfont}{\normalsize\itshape}        % subsection font size

% Make chapter, section, and subsection entries uppercase
\renewcommand{\cftchappresnum}{\MakeUppercase}
\renewcommand{\cftchapaftersnum}{}
\let\oldcontentsline\contentsline
\renewcommand{\contentsline}[4]{%
  \ifnum\pdfstrcmp{#1}{chapter}=0
  \oldcontentsline{#1}{\MakeUppercase{#2}}{#3}{#4}%
  \else\ifnum\pdfstrcmp{#1}{section}=0
  \oldcontentsline{#1}{\MakeUppercase{#2}}{#3}{#4}%
  \else
  \oldcontentsline{#1}{#2}{#3}{#4}%
  \fi\fi
}

% Optional: adjust indentation
\setlength{\cftsecindent}{1em}      % section indentation
\setlength{\cftsubsecindent}{2em}     % subsection indentation

% Optional: add dots for chapters
\renewcommand{\cftchapleader}{\cftdotfill{\cftdotsep}}
\renewcommand{\cftdotsep}{2}

% Add period after chapter, section, and subsection numbers
\renewcommand{\cftchapaftersnum}{.}
\renewcommand{\cftsecaftersnum}{.}
\renewcommand{\cftsubsecaftersnum}{.}

  \fancyhf{}
  \renewcommand{\headrulewidth}{0pt}
  \renewcommand{\footrulewidth}{0pt}
  \fancyfoot[R]{\thepage}
}
